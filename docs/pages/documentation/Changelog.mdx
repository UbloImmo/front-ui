import { Meta, Markdown } from "@storybook/blocks";
import { Hypertext } from "@components";
import { HeaderInfo, Text } from "../../blocks";
import { Content, Header } from "../../containers";
import changelog from "../../../CHANGELOG.md?raw";


<Meta title="Documentation/Changelog"/>

<Header>
  <HeaderInfo title="Changelog" parent="Documentation">
    <Text size="m" color="primary-dark" important>
    All notable changes to the library will be documented in this file.<br/>
    The format is based on <Hypertext href="https://keepachangelog.com/en/1.1.0/">Keep a Changelog</Hypertext>,&nbsp;
    and this project adheres to <Hypertext href="https://semver.org/spec/v2.0.0.html">Semantic Versioning</Hypertext>.
    </Text>
  </HeaderInfo>
</Header>

<Content>

<Markdown>{(() => {
  const contentIndex = changelog.indexOf("##");

  const headingRegex = /^##\s[\w\s\d.-]+\S$/gim;
  const hr = "<hr />";
  return changelog
    .substring(contentIndex)
    .replaceAll("https://ublo.ublo.org/design-system", "")
    .replaceAll("###", "####")
    .replaceAll(headingRegex, (match) => `${match}\n${hr}`);
})()}</Markdown>

</Content>
